Installation

Get started with LumeUI in just a few steps. Import the components you need and start building.

HTML 
<!-- Include LumeUI components -->
<script src="path/to/lume-components.js"></script>

<!-- Or import specific components -->
<script type="module">
  import { LumeButton, LumeCard } from './lume-components.js';
</script>

Quick Start

HTML 
<!-- Your first LumeUI component -->
<lume-button variant="primary">Click me!</lume-button>

<lume-card>
  <h3 slot="header">Welcome</h3>
  <p>This is your first LumeUI card component.</p>
</lume-card>
